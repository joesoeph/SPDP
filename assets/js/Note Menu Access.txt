CREATE TABLE `menuaccess` (
	`MenuAccessId` INT(2) NOT NULL AUTO_INCREMENT,
	`UsersId` INT(5) NULL DEFAULT NULL,
	`MenuId` INT(2) NULL DEFAULT NULL,
	`Read` INT(1) NULL DEFAULT NULL,
	`Write` INT(1) NULL DEFAULT NULL,
	`Update` INT(1) NULL DEFAULT NULL,
	`Delete` INT(1) NULL DEFAULT NULL,
	PRIMARY KEY (`MenuAccessId`)
)
ENGINE=InnoDB
;

CREATE TABLE `menu` (
	`MenuId` INT(11) NOT NULL AUTO_INCREMENT,
	`MenuText` VARCHAR(50) NULL DEFAULT NULL,
	`MenuIcon` VARCHAR(100) NULL DEFAULT NULL,
	`Parent` INT(11) NULL DEFAULT NULL,
	`Active` INT(11) NULL DEFAULT NULL,
	PRIMARY KEY (`MenuId`)
)
ENGINE=InnoDB
;
